#include <stdio.h>  
#define PLOY  0xa001 //0x8005·´×ª  
  
unsigned short crc16(char *data,int size)  
{  
    unsigned short crc = 0x0;  
    int flag; 
    char data_t ; 
    int i = 0;  
    int j = 0;    
    if(data ==NULL){  
        return 0;  
    }  
    for(j=0;j<size;j++){  
        data_t = *data++;  
        crc = (data_t ^ (crc));  
        for(i=0;i<8;i++){  
            if((crc&0x1)==1){  
                crc = (crc>>1) ^ PLOY;  
            }else{  
                crc >>= 1;  
            }  
        }  
    }  
      
    return crc;  
}  
  
int main(int argc,char **argv)  
{  
    char a[]={1,2,3,4,5,6};  
    printf("%x\n",crc16(a,sizeof(a)/sizeof(char)));  
    return 0;  
}
